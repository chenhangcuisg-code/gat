# ---- GDScript that would be sent ----
@tool
extends RefCounted
# instantiate_scene — instance packed scene "res://ui/Hud.tscn" under "." as "Hud"

func execute(executeContext):
	var ei = executeContext.editor_plugin.get_editor_interface()
	var root = ei.get_edited_scene_root()
	if root == null:
		executeContext.output("error", "no scene is open")
		return
	if not ResourceLoader.exists("res://ui/Hud.tscn"):
		executeContext.output("error", "scene not found: res://ui/Hud.tscn")
		return
	var parent = root if "." == "." else root.get_node_or_null(".")
	if parent == null:
		executeContext.output("error", "parent not found: .")
		return
	var inst = load("res://ui/Hud.tscn").instantiate()
	inst.name = "Hud"
	parent.add_child(inst)
	inst.owner = root
	executeContext.output("instanced", str(root.get_path_to(inst)))
